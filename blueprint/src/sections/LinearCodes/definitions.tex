\section{Definitions}
\begin{definition}[Linear Code by Subspace]
\label{def:LC_subspace}
Let $q$ be a prime power (i.e. $q = p^{s}$ for some prime $p$ and integer $s \geq 1$).
$C \subseteq \mathbb{F}^{n}_{q}$ is a linear code if it is a linear subspace of $\mathbb{F}^{n}_{q}$.
If $C$ has dimenension $k$ and distance $d$ then it will be referred to as an $[n,k,d]_{q}$ or just an $[n,k]_{q}$ code.
\end{definition}

\begin{definition}[Generator Matrix]
\label{def:GeneratorMatrix}
\lean{GeneratorMatrix}
\leanok
\uses{def:LC_subspace}
If $C$ is an $[n, k]_{q}$ linear code then there exists a matrix $G \in \mathbb{F}^{k \times n}_{q}$ of rank $k$ satisfying
\[C = \{x \cdot G \mid x \in \mathbb{F}^{k}_{q}.\}\]
We define $G$ as a generator matrix of $C$.
Note that $G$ is not unique.
\end{definition}

\begin{definition}[Parity Check Matrix]
\label{def:ParityCheckMatrix}
\lean{ParityCheckMatrix}
\leanok
\uses{def:LC_subspace}
If $C$ is an $[n, k]_{q}$ linear code then there exists a matrix $H \in \mathbb{F}^{(n-k) \times n}_{q}$ of rank $n - k$ satisfying
\[C = \{\mathbf{y} \in \mathbb{F}^{n}_{q} \mid H \cdot \mathbf{y}^{\intercal} = 0\}.\]
We define $H$ as a partiy check matrix of $C$.
Note that $H$ is not unique.
\end{definition}
