\section{Lemmas}

\begin{lemma}[Subspace Implies Generator]
    \uses{def:LC_subspace,def:GeneratorMatrix}
    \label{lemma:subspace-imp-generator}
    Let $C$ be a linear code that is a $k$-dimensional subspace of $\mathbb{F}^{n}_{q}$.
    Then there exists a $k \times n$ matrix $G$ over $\mathbb{F}_q$ such that
    \[
        C = \{\mathbf{f}G \mid \mathbf{f} \in \mathbb{F}^{k}_{q}\}.
    \]
\end{lemma}

\begin{proof}
Note that $C$ is a $k$-dimensional subspace of $\mathbb{F}_q^n$, and thus has a basis $\{\mathbf{g}_1, \ldots, \mathbf{g}_k\}$.
Let $G$ be the $k \times n$ matrix whose $i$-th row is $\mathbf{g}_i$.
For any $\mathbf{f} = (f_1, \ldots, f_k) \in \mathbb{F}_q^k$, the row-vector product is
\[
    \mathbf{f}G = \sum_{i=1}^k f_i \mathbf{g}_i.
\]
Thus $\{\mathbf{f}G \mid f \in \mathbb{F}^{k}_{q}\}$ is the set of $\mathbb{F}_q$-linear combinations of $\mathbf{g}_{1},\ldots, \mathbf{g}_{k}$.
Since $\mathbf{g}_{1}, \ldots, \mathbf{g}_{k}$ is a basis of $C$ and the span equals $C$.
We must have $C = \{\mathbf{f}G \mid f \in \mathbb{F}^{k}_{q}\}$.
\end{proof}

\begin{lemma}[Generator Implies Subspace]
    \uses{def:LC_subspace, def:GeneratorMatrix}
    \label{lemma:generator-imp-subspace}
    Let $G \in \mathbb{F}^{k \times n}_{q}$ and let $C = \{\mathbf{f}G \mid \mathbf{f} \in \mathbb{F}^{k}_{q}\}$.
    Then $C$ is a subspace of $\mathbb{F}^{n}_{q}$.
\end{lemma}

\begin{proof}
    First of all, we know $C\subseteq \mathbb{F}^{n}_{q}$
        because $\forall c\in C$, $G$ constrains $c$ to be $n$ dimensional. 
        Additionally, because of the closure of $\mathbb{F}_q$ under addition and multiplication that $c\in \mathbb{F}^{n}_{q}$ and therefore $C\subseteq \mathbb{F}^{n}_{q}$. Given this, it should be sufficient to prove that $C$ contains the zero vector and is closed under linear addition and scalar multiplication
    \begin{itemize}
        \item We can show that the zero vector $\vec{0}\in \mathbb{F}^{n}_{q}$ is in $C$ because $\vec{0}G=\vec{0}$.
        \item We can show that $C$ is closed under addition by considering that $\forall a,b\in C$,
            $\exists x,y\in \mathbb{F}^{k}_{q}$ such that $a=xG,b=yG$, 
            distributivity of matrix multiplication gives us $a+b=xG+yG=(x+y)G$, 
            and since $x+y\in \mathbb{F}^{k}_{q}$ because of closure under addition, 
            $a+b=(x+y)G\in C $
        \item We can show that $C$ is closed under scalar multiplication by considering that $\forall a\in C$ that
            $\exists x\in \mathbb{F}^{k}_{q}$ such that $a=xG$.
            And therefore, $\forall d\in \mathbb{F}_{q}, da=dxG=(dx)G$. 
            And since $dx\in \mathbb{F}^{k}_{q}$ by its own closure under scalar multiplication it must follow $d(xG)\in C$
        These three conditions should be sufficient to prove that $C$ is a valid vector space.
    \end{itemize}
    Therefore $C$ is a subspace of $\mathbb{F}^{n}_{q}$
\end{proof}

\begin{lemma}[Subspace implies Parity Check]
    \uses{def:LC_subspace,def:ParityCheckMatrix,lemma:subspace-imp-generator}
    \label{lemma:subspace-imp-parity-check}
    Let $C$ be a linear code of dimension $k$ that is a subspace of $\mathbb{F}^{n}_{q}$.
    Then there exists a matrix $H \in \mathbb{F}^{(n-k) \times n}_{q}$ such that
    \[
        C = \{\mathbf{f} \in \mathbb{F}^{n}_{q} \mid H\mathbf{f}^{\intercal} = \mathbf{0}\}.
    \]
\end{lemma}

\begin{proof}
    We can use the Subspace Implies Generator lemma to say that we know that $\exists G, \forall c \in C, \exists x\in \mathbb{F}^{k}_{q}, c=xG$. We can extend this further to say that because:
    \begin{itemize}
        \item The rows of $G$ are linearly independent since they are basis vectors
        \item $G$ has $k$ rows
        \item $G$ is not necessarily unique for any given $C$
        \item We are given (https://www.cfm.brown.edu/people/dobrush/cs52/Mathematica/Part3/rank.html) that $G$ must have $k$ linearly independent columns since its column rank equals its row rank of $k$
    \end{itemize}
    We can use Gaussian elimination to attain standard form $G'=[I_k\mid P]$ where $P$ is some $k \times (n-k)$ submatrix
        More formally,
        \[
            \exists G, \forall c \in C, \exists x\in \mathbb{F}^{k}_{q}, c=xG, \forall i,j \leq k , i\neq j \implies G_{ij} = 0, i=j \implies G_{ij}=1
        \]
    (need to find if there's a good theorem to use in lean, I found a more elaborate exmplanation of the above statement here
        `https://math.stackexchange.com/questions/4166013/why-are-the-codes-generated-by-g-and-i-k-mid-a-equivalent`)
    \\\\
    Then suppose we take $H=[-P^T\mid I_{n-k}]$. I.e. the first $k$ columns of $H$ are going to be the rows of $P$ multiplied by -1.
    \\\\
    Note that this preserves the $n-k$ rows requirement since $P$ has $n-k$ columns, and similarly, the right hand side of the matrix is the $n-k x n-k$ identity matrix. We can then see that 
    \[HG^T=I_k\cdot -P^T + I_{n-k}\cdot P^t= -P^T+P^t=0^{(n-k) x k}\] i.e. an $(n-k) x k$ with only zeroes as entries
    \\\\
        To make this a little bit more detailed, we can first get that $HG^T$ is a $(n-k) \times k$ matrix, since it's the row count of $H$ and $G$.
        If we look at matrix multiplication in terms of the dot products of rows of $H$ and columns of $G^T$,
        Essentially meaning that each entry in $HG^T$ is some dot product of a row of $H$ and a row of $G$
        More formally $\forall i \in [n-k], \forall j \in [k], (HG^T)_{ij}$ is the dot product of the $ith$ row of $H$ and the $jth$ row of $G$. Which will then be the $jth$ element of the $ith$ row of $-P^T$ or effectively $-1*P^T_{ij}=-1*P_{ji}$ plus the $ith$ element of the $jth$ row of $P$ or $P_{ji}$ which means 
        \[\forall i \in [n-k], \forall j \in [k], (HG^T)_{ij}=P_{ji}-P_{ji}=0\]
    Using this we can make the claim that $\forall x \in \mathbb{F}^{k}_{q}, H(xG)^T=\mathbf{0}$ by showing
        $\forall x \in \mathbb{F}^{k}_{q} H(xG)^T=HG^Tx^t$ from distributing the transverse. And then, from our statement above, we have that \[H(xG)^T=HG^Tx^t=0^{(n-k) x k}x^t=\mathbf{0}\]
    And, from the basic definition of $G$ $c\in C \iff \exists x \in \mathbb{F}^{k}_{q}, c=xG$
    Which we can use our equivalence to say $c\in C \iff \exists x \in \mathbb{F}^{k}_{q}, Hc^T=\mathbf{0}$ yielding that
    \[C \subseteq \{\mathbf{f} \in \mathbb{F}^{n}_{q} \mid H\mathbf{f}^{\intercal} = \mathbf{0}\}\]
    To make notation easier, let's say this $H$ generated subspace (we know it's a subspace because of Parity Check Implies subspace below) is $C'$. We know that 
    \begin{itemize}
        \item $dim({F}^{n}_{q})=n$
        \item $dim(C)=k$
    \end{itemize}
    We can assert that the rank of $H$ is $n-k$ because the identity matrix component on the right hand side forces the rows to be linearly independent. From here we can use the rank-nullity theorem to say that, since $C'$ represents the null space of $H$
    \[dim(C')+rank(H)=dim({F}^{n}_{q})\]
    \[\implies dim(C') + n-k = n\]
    \[\implies dim(C') = k = dim(C)\]
    and since $dim(C')=dim(C)$ and $C\subseteq C'$, it should follow that $C=C'$
\end{proof}

\begin{lemma}[Parity Check Implies subspace]
    \uses{def:LC_subspace, def:ParityCheckMatrix}
    \label{lemma:parity-check-imp-subspace}
    Let $H \in \mathbb{F}^{(n-k) \times n}_{q}$ and let $C = \{\mathbf{f} \in \mathbb{F}^{n}_{q} \mid H\mathbf{f}^{\intercal} = \mathbf{0}\}$.
    Then $C$ is a subspace of $\mathbb{F}^{n}_{q}$.
\end{lemma}

\begin{proof}
    First we can show that $C \subseteq \mathbb{F}^{n}_{q}$ because $C$ every element of $C$ is already defined as being in $\mathbb{F}^{n}_{q}$. Given this, it should be sufficient to prove that $C$ contains the zero vector and is closed under linear addition and scalar multiplication
    \begin{itemize}
        \item We can show that the zero vector $\vec{0}\in \mathbb{F}^{n-k}_{q}$ is in $C$ because $H\vec{0}^T=\vec{0}$
        \item We can show that $C$ is closed under addition by considering that $\forall a,b\in C, Ha^T=Hb^T=0$, meaning $Ha^T+Hb^T= 0+0 = 0$
            and therefore by distributivity of matrix multiplication and distributivity of transverses, We have that 
            \[Ha^T+Hb^T = H(a^T+b^T) = H(a+b)^T = 0\]
            and since $(a+b)\in \mathbb{F}^{n}_{q}$ by closure of $\mathbb{F}^{n}_{q}$ under addition, $H(a+b)^T = 0$, and $a+b\in C$
        \item We can show that $C$ is closed under scalar multiplication by considering that $\forall a\in C$ that
            $Ha^T=0$, then $\forall d\in \mathbb{F}_{q}$, that $Ha^T=0 \implies dHa^T = d*0=0$
            which $\implies H(da)^T==0$ because we can commute scalars with matrices and vectors,
            giving us that $da\in C$
    \end{itemize}
    These three conditions should be sufficient to prove that $C$ is a valid vector space.
\end{proof}
