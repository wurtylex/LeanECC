\begin{lemma}[subspace-imp-generator]
    \uses{def:LC_subspace,def:GeneratorMatrix}
    \label{lemma:subspace-imp-generator}
    Let $C$ be a linear code that is a $k$-dimensional subspace of $\mathbb{F}^{n}_{q}$.
    Then there exists a $k \times n$ matrix $G$ over $\mathbb{F}_q$ such that
    \[
        C = \{\mathbf{f}G \mid \mathbf{f} \in \mathbb{F}^{k}_{q}\}.
    \]
\end{lemma}

\begin{proof}
Note that $C$ is a $k$-dimensional subspace of $\mathbb{F}_q^n$, and thus has a basis $\{\mathbf{g}_1, \ldots, \mathbf{g}_k\}$.
Let $G$ be the $k \times n$ matrix whose $i$-th row is $\mathbf{g}_i$.
For any $\mathbf{f} = (f_1, \ldots, f_k) \in \mathbb{F}_q^k$, the row-vector product is
\[
    \mathbf{f}G = \sum_{i=1}^k f_i \mathbf{g}_i.
\]
Thus $\{\mathbf{f}G \mid f \in \mathbb{F}^{k}_{q}\}$ is the set of $\mathbb{F}_q$-linear combinations of $\mathbf{g}_{1},\ldots, \mathbf{g}_{k}$.
Since $\mathbf{g}_{1}, \ldots, \mathbf{g}_{k}$ is a basis of $C$ and the span equals $C$.
We must have $C = \{\mathbf{f}G \mid f \in \mathbb{F}^{k}_{q}\}$.
\end{proof}

\begin{lemma}[generator-imp-subspace]
    \uses{def:LC_subspace, def:GeneratorMatrix}
    \label{lemma:generator-imp-subspace}
    Let $G \in \mathbb{F}^{k \times n}_{q}$ and let $C = \{\mathbf{f}G \mid \mathbf{f} \in \mathbb{F}^{k}_{q}\}$.
    Then $C$ is a subspace of $\mathbb{F}^{n}_{q}$.
\end{lemma}

\begin{proof}
    First of all, we know $C\subseteq \mathbb{F}^{n}_{q}$
        because $\forall c\in C$, we know that $G$ constrains $c$ to be $n$ dimensional
        Additionally, because of the closure of $\mathbb{F}_q$ under addition and multiplication
        that then $c\in \mathbb{F}^{n}_{q}$ and therefore $C\subseteq \mathbb{F}^{n}_{q}$
        Given this, it should be sufficient to prove that $C$ contains the zero vector and is closed under linear addition and scalar multiplication
    We can show that the zero vector $\vec{0}\in \mathbb{F}^{n}_{q}$ is in $C$ because $\vec{0}G=\vec{0}$
        therefore $\forall G$, $\vec{0}\in C$
    We can show that $C$ is closed under addition by considering that $\forall a,b\in C$,
        $\exists x,y\in \mathbb{F}^{k}_{q}$ such that $a=xG,b=yG$
        and from distributivity of matrix multiplication that $a+b=xG+yG=(x+y)G$
        and since $x+y\in \mathbb{F}^{k}_{q}$ because of its closure under addition, this means
        $a+b=(x+y)G\in C $
    We can show that $C$ is closed under scalar multiplication by considering that $\forall a\in C$ that
        $\exists x\in \mathbb{F}^{k}_{q}$ such that $a=xG$
        and that therefore $\forall d\in \mathbb{F}_{q}, da=dxG=(dx)G$
        and since $dx\in \mathbb{F}^{k}_{q}$ by its own closure under scalar multiplication it must follow $d(xG)\in C$
    These three conditions should be sufficient to prove that $C$ is a valid vector space.
    Therefore $C$ is a subspace of $\mathbb{F}^{n}_{q}$
\end{proof}

\begin{lemma}[subspace-imp-parity-check]

    Then there exists a matrix $H \in \mathbb{F}^{(n-k) \times n}_{q}$ such that
    \[
        C = \{\mathbf{f} \in \mathbb{F}^{n}_{q} \mid H\mathbf{f}^{\intercal} = \mathbf{0}\}.
    \]
\end{lemma}

\begin{proof}

    $c\in C, \iff \exists x, c=xG \iff Hc^T=H(xG)^T=HG^Tx^t=0x^t=0$, meaning that $c\in C \iff Hc^T = 0$

\end{proof}

\begin{lemma}[parity-check-imp-subspace]
    \uses{def:LC_subspace, def:ParityCheckMatrix}
    \label{lemma:parity-check-imp-subspace}
    Let $H \in \mathbb{F}^{(n-k) \times n}_{q}$ and let $C = \{\mathbf{f} \in \mathbb{F}^{n}_{q} \mid H\mathbf{f}^{\intercal} = \mathbf{0}\}$.
    Then $C$ is a subspace of $\mathbb{F}^{n}_{q}$.
\end{lemma}

\begin{proof}
    First we can show that $C \subseteq \mathbb{F}^{n}_{q}$ because $C$ every element of $C$ is already defined as being in $\mathbb{F}^{n}_{q}$
        Given this, it should be sufficient to prove that $C$ contains the zero vector and is closed under linear addition and scalar multiplication
    We can show that the zero vector $\vec{0}\in \mathbb{F}^{n-k}_{q}$ is in $C$ because $H\vec{0}^T=\vec{0}$
    We can show that $C$ is closed under addition by considering that $\forall a,b\in C, Ha^T=Hb^T=0$
        This also gives us that $Ha^T+Hb^T= 0+0 = 0$
        and therefore by distributivity of matrix multiplication, We have that $Ha^T+Hb^T = H(a^T+b^T) = 0$
        and by the distributivity of transverses, that $Ha^T+Hb^T = H(a^T+b^T) = H(a+b)^T = 0$
        and since $(a+b)\in \mathbb{F}^{n}_{q}$ by closure of $\mathbb{F}^{n}_{q}$ under addition,
        and because $H(a+b)^T = 0$, $a+b\in C$
    We can show that $C$ is closed under scalar multiplication by considering that $\forall a\in C$ that
        $Ha^T=0$, then $\forall d\in \mathbb{F}_{q}$, that $Ha^T=0 \implies dHa^T = d*0=0$
        which $\implies H(da)^T==0$ because we can commute scalars with matrices and vectors,
        giving us that $da\in C$
    These three conditions should be sufficient to prove that $C$ is a valid vector space.
\end{proof}
