\section{Properties}
\begin{lemma}[Dimension is at most the block length]
    \label{lem:dim_le_n}
    \lean{Code.dim_le_n}
    \leanok
    \uses{def:dim}
    For every code $C$ of block length $n$, we have $k \leq n$.
\end{lemma}

\begin{proof}
    \leanok
    \uses{def:q}
    Note that $C \subseteq \alpha^{n}$, we have $\lvert C\rvert \le \lvert\alpha\rvert^{n} = q^{n}$, thus $k = \log_{q}\lvert C\rvert \le \log_{q} q^{n} = n$.
\end{proof}
 
\begin{lemma}[Rate is at most one]
    \label{lem:rate_le_one}
    \lean{Code.rate_le_one}
    \leanok
    \uses{def:rate}
    For every code $C$, we have $R \leq 1$.
\end{lemma}
\begin{proof}
    \leanok
    \uses{lem:dim_le_n}
    By Lemma~\ref{lem:dim_le_n}, $k \leq n$, and $n \ge 0$.
    Therefore $R = k/n \leq 1$.
\end{proof}

\begin{lemma}[Sphere cardinality]
  \label{lemma:SphereCard}
  \uses{def:q}
  For any $\mathbf{x} \in [q]^n$ and integer $i \geq 0$,
  \[\bigl|\{\mathbf{y} \in [q]^n \mid \Delta(\mathbf{x}, \mathbf{y}) = i\}\bigr| = \binom{n}{i}(q-1)^i.\]
\end{lemma}

\begin{proof}
  Write $S_i = \{\mathbf{y} \in [q]^n \mid \Delta(\mathbf{x}, \mathbf{y}) = i\}$ for the sphere,
  and for a word $\mathbf{y} \in [q]^n$ let
  \[
    T(\mathbf{y}) = \{j \in [n] \mid y_j \neq x_j\}
  \]
  be its set of disagreement coordinates. By the definition of Hamming distance,
  $\Delta(\mathbf{x}, \mathbf{y}) = |\{j \in [n] \mid x_j \neq y_j\}| = |T(\mathbf{y})|$, so
  membership $\mathbf{y} \in S_i$ is the same as the condition $|T(\mathbf{y})| = i$.

  We construct an explicit bijection. Let $\mathcal{P}_i$ be the set of pairs $(T, g)$ in which $T \subseteq [n]$
  has $|T| = i$ and $g : T \to [q]$ satisfies $g(j) \neq x_j$ for every $j \in T$. Define the
  forward map
  \[
    \Phi : S_i \to \mathcal{P}_i, \qquad
    \Phi(\mathbf{y}) = \bigl(T(\mathbf{y}),\ (y_j)_{j \in T(\mathbf{y})}\bigr);
  \]
  this is well defined because $|T(\mathbf{y})| = i$ by the paragraph above, and $y_j \neq x_j$
  for each $j \in T(\mathbf{y})$ by the definition of $T(\mathbf{y})$. Define the reverse map
  \[
    \Psi : \mathcal{P}_i \to S_i, \qquad
    \Psi(T, g) = \mathbf{y}, \quad\text{where } y_j =
      \begin{cases} g(j) & \text{if } j \in T, \\ x_j & \text{if } j \notin T. \end{cases}
  \]
  Its image lies in $S_i$: the word $\mathbf{y} = \Psi(T, g)$ has $y_j \neq x_j$ exactly on $T$
  (on $T$ since $g(j) \neq x_j$, and off $T$ since $y_j = x_j$), so $T(\mathbf{y}) = T$ and
  therefore $|T(\mathbf{y})| = |T| = i$. The two maps are mutually inverse:
  \begin{itemize}
    \item $\Psi(\Phi(\mathbf{y})) = \mathbf{y}$: on $T(\mathbf{y})$ the reconstructed word takes
      the value $y_j$, and off $T(\mathbf{y})$ it takes the value $x_j$, which equals $y_j$
      because $j \notin T(\mathbf{y})$ means exactly that $y_j = x_j$.
    \item $\Phi(\Psi(T, g)) = (T, g)$: we showed $T(\Psi(T, g)) = T$, and on $T$ the values kept
      by $\Phi$ are the $g(j)$.
  \end{itemize}
  Therefore, $|S_i| = |\mathcal{P}_i|$.

  It remains to count $\mathcal{P}_i$. Grouping the pairs by their first coordinate $T$,
  \[
    |\mathcal{P}_i| = \sum_{\substack{T \subseteq [n] \\ |T| = i}}
      \bigl|\{\, g : T \to [q] \mid g(j) \neq x_j \text{ for all } j \in T \,\}\bigr|.
  \]
  For a fixed $T$, specifying such a $g$ means choosing, independently for each $j \in T$, one of
  the $q - 1$ values in $[q] \setminus \{x_j\}$; there are thus $(q-1)^{|T|} = (q-1)^i$ such
  functions. As the number of $i$-element subsets of $[n]$ is $\binom{n}{i}$,
  \[
    |S_i| = |\mathcal{P}_i|
      = \sum_{\substack{T \subseteq [n] \\ |T| = i}} (q-1)^i
      = \binom{n}{i}(q-1)^i. \qedhere
  \]
\end{proof}

\begin{lemma}[Ball cardinality equals volume]
  \label{lemma:BallCard}
  \lean{Code.ncard_hammingBall}
  \leanok
  \uses{def:HammingBall, def:Vol, def:q}
  For any $\mathbf{x} \in [q]^n$ and $r \geq 0$,
  \[|B(\mathbf{x}, r)| = \Vol_q(n, r).\]
\end{lemma}

\begin{proof}
  \uses{lemma:SphereCard}
  For each integer $0 \leq i \leq r$ let $S_i = \{\mathbf{y} \in [q]^n \mid \Delta(\mathbf{x}, \mathbf{y}) = i\}$
  be the sphere of radius $i$ about $\mathbf{x}$.

  The $S_i$ are pairwise disjoint: if $\mathbf{y} \in S_i \cap S_j$ then
  $i = \Delta(\mathbf{x}, \mathbf{y}) = j$, so distinct indices give disjoint spheres, as a word
  has a single Hamming distance to $\mathbf{x}$.

  The $S_i$ also cover the ball. By the definition of the Hamming ball,
  $\mathbf{y} \in B(\mathbf{x}, r)$ means $\Delta(\mathbf{x}, \mathbf{y}) \leq r$. Since
  $\Delta(\mathbf{x}, \mathbf{y})$ is a nonnegative integer, this holds iff
  $\Delta(\mathbf{x}, \mathbf{y}) = i$ for exactly one $i$ with $0 \leq i \leq r$, i.e. iff
  $\mathbf{y} \in S_i$ for some such $i$. Therefore,
  \[
    B(\mathbf{x}, r) = \bigsqcup_{i=0}^{r} S_i,
  \]
  a disjoint union by the previous paragraph.

  Finally, we compute the cardinalities. Because the union is disjoint, its cardinality is the sum of the
  cardinalities of the parts, and each part is counted by Lemma~\ref{lemma:SphereCard}:
  \[
    |B(\mathbf{x}, r)|
      = \sum_{i=0}^{r} |S_i|
      = \sum_{i=0}^{r} \binom{n}{i}(q-1)^i.
  \]
  The right-hand side is exactly $\Vol_q(n, r)$ by the definition of the volume, so
  $|B(\mathbf{x}, r)| = \Vol_q(n, r)$.
\end{proof}

\begin{lemma}[Volume entropy bound]
\label{lemma:VolEntropy}
\lean{Code.hammingVolume_le_pow_mul_entropy}
\leanok
\uses{def:q, def:Vol}
Let $q \geq 2$ and $n \geq 1$ be integers, and let $r$ be an integer with $0 \leq r$ and $\frac{r}{n} \leq 1 - \frac{1}{q}$. Then
\[
  \Vol_q(r, n) \leq q^{\,n H_q(r/n)}.
\]
\end{lemma}
\begin{proof}
  Write $\lambda = \frac{r}{n}$. If $r = 0$ then $\Vol_q(0, n) = 1 = q^{n H_q(0)}$, so assume $r \geq 1$, giving $0 < \lambda \leq 1 - \frac{1}{q} < 1$. From the definition of $H_q$,
  \[
    q^{n H_q(\lambda)} = (q-1)^{r}\, \lambda^{-r}\,(1-\lambda)^{-(n-r)},
  \]
  so it suffices to show $\Vol_q(r,n)\,(q-1)^{-r}\lambda^{r}(1-\lambda)^{n-r} \leq 1$. Put $\theta = \frac{\lambda}{(q-1)(1-\lambda)}$. Note that $\lambda \leq 1 - \frac{1}{q}$ gives $\frac{\lambda}{1-\lambda} \leq q-1$, hence $0 \leq \theta \leq 1$. For each $0 \leq i \leq r$, since $r - i \geq 0$ and $\theta \leq 1$,
  \[
    \binom{n}{i} (q-1)^{i} (q-1)^{-r}\lambda^{r}(1-\lambda)^{n-r} = \binom{n}{i} \lambda^{i} (1-\lambda)^{n-i}\,\theta^{\,r-i} \leq \binom{n}{i} \lambda^{i} (1-\lambda)^{n-i}.
  \]
  Summing over $0 \leq i \leq r$ and extending the sum to $0 \leq i \leq n$,
  \[
    \Vol_q(r,n)\,(q-1)^{-r}\lambda^{r}(1-\lambda)^{n-r} \leq \sum_{i=0}^{n}\binom{n}{i} \lambda^{i}(1-\lambda)^{n-i} = (\lambda + (1-\lambda))^{n} = 1,
  \]
  the last step by the binomial theorem. Rearranging gives $\Vol_q(r,n) \leq q^{n H_q(\lambda)}$.
\end{proof}
