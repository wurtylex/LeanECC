\section{Properties}
\begin{lemma}[Dimension is at most the block length]
    \label{lem:dim_le_n}
    \lean{Code.dim_le_n}
    \leanok
    \uses{def:dim}
    For every code $C$ of block length $n$, we have $k \leq n$.
\end{lemma}

\begin{proof}
    \leanok
    \uses{def:q}
    Note that $C \subseteq \alpha^{n}$, we have $\lvert C\rvert \le \lvert\alpha\rvert^{n} = q^{n}$, thus $k = \log_{q}\lvert C\rvert \le \log_{q} q^{n} = n$.
\end{proof}
 
\begin{lemma}[Rate is at most one]
    \label{lem:rate_le_one}
    \lean{Code.rate_le_one}
    \leanok
    \uses{def:rate}
    For every code $C$, we have $R \leq 1$.
\end{lemma}
\begin{proof}
    \leanok
    \uses{lem:dim_le_n}
    By Lemma~\ref{lem:dim_le_n}, $k \leq n$, and $n \ge 0$.
    Therefore $R = k/n \leq 1$.
\end{proof}

\begin{lemma}[Sphere cardinality]
  \label{lemma:SphereCard}
  \uses{def:q}
  For any $\mathbf{x} \in [q]^n$ and integer $i \geq 0$,
  \[\bigl|\{\mathbf{y} \in [q]^n \mid \Delta(\mathbf{x}, \mathbf{y}) = i\}\bigr| = \binom{n}{i}(q-1)^i.\]
\end{lemma}

\begin{proof}
  Let $S_i = \{\mathbf{y} \in [q]^n \mid \Delta(\mathbf{x}, \mathbf{y}) = i\}$, and for
  $\mathbf{y} \in [q]^n$ set $T(\mathbf{y}) = \{j \in [n] \mid y_j \neq x_j\}$. For
  $\mathbf{y} \in S_i$:
  \begin{itemize}
    \item $|T(\mathbf{y})| = \Delta(\mathbf{x}, \mathbf{y}) = i$;
    \item $y_j \in [q] \setminus \{x_j\}$ for $j \in T(\mathbf{y})$, and $y_j = x_j$ for
      $j \notin T(\mathbf{y})$.
  \end{itemize}
  Conversely, any $i$-element subset $T \subseteq [n]$ together with any choice of a symbol
  $y_j \in [q] \setminus \{x_j\}$ for each $j \in T$ (and $y_j = x_j$ elsewhere) yields a
  distinct word at distance exactly $i$ from $\mathbf{x}$. This gives a bijection between the
  sphere and these pairs. There are $\binom{n}{i}$ choices of the subset $T$, and for each one
  $(q-1)^i$ choices of the symbols, so the sphere has $\binom{n}{i}(q-1)^i$ elements.
\end{proof}

\begin{lemma}[Ball cardinality equals volume]
  \label{lemma:BallCard}
  \uses{def:HammingBall, def:Vol, def:q}
  For any $\mathbf{x} \in [q]^n$ and $r \geq 0$,
  \[|B(\mathbf{x}, r)| = \Vol_q(n, r).\]
\end{lemma}

\begin{proof}
  \uses{lemma:SphereCard}
  For $0 \leq i \leq r$ let $S_i = \{\mathbf{y} \mid \Delta(\mathbf{x}, \mathbf{y}) = i\}$.
  Every word $\mathbf{y}$ has a single Hamming distance to $\mathbf{x}$, so all of the $S_i$ are
  pairwise disjoint, and since $\Delta(\mathbf{x}, \mathbf{y}) \leq r$ holds iff
  $\Delta(\mathbf{x}, \mathbf{y}) = i$ for some $0 \leq i \leq r$, we have
  $B(\mathbf{x}, r) = \bigsqcup_{i=0}^{r} S_i$. Therefore,
  \[
    |B(\mathbf{x}, r)| = \sum_{i=0}^{r} |S_i| = \sum_{i=0}^{r} \binom{n}{i}(q-1)^i = \Vol_q(n, r),
  \]
  using Lemma~\ref{lemma:SphereCard} for each sphere and the definition of $\Vol_q$ for the
  last equality.
\end{proof}

\begin{lemma}[Volume entropy bound]
\label{lemma:VolEntropy}
\lean{Code.hammingVolume_le_pow_mul_entropy}
\uses{def:q, def:Vol}
Let $q \geq 2$ and $n \geq 1$ be integers, and let $r$ be an integer with $0 \leq r$ and $\frac{r}{n} \leq 1 - \frac{1}{q}$. Then
\[
  \Vol_q(r, n) \leq q^{\,n H_q(r/n)}.
\]
\end{lemma}
\begin{proof}
  Write $\lambda = \frac{r}{n}$. If $r = 0$ then $\Vol_q(0, n) = 1 = q^{n H_q(0)}$, so assume $r \geq 1$, giving $0 < \lambda \leq 1 - \frac{1}{q} < 1$. From the definition of $H_q$,
  \[
    q^{n H_q(\lambda)} = (q-1)^{r}\, \lambda^{-r}\,(1-\lambda)^{-(n-r)},
  \]
  so it suffices to show $\Vol_q(r,n)\,(q-1)^{-r}\lambda^{r}(1-\lambda)^{n-r} \leq 1$. Put $\theta = \frac{\lambda}{(q-1)(1-\lambda)}$. Note that $\lambda \leq 1 - \frac{1}{q}$ gives $\frac{\lambda}{1-\lambda} \leq q-1$, hence $0 \leq \theta \leq 1$. For each $0 \leq i \leq r$, since $r - i \geq 0$ and $\theta \leq 1$,
  \[
    \binom{n}{i} (q-1)^{i} (q-1)^{-r}\lambda^{r}(1-\lambda)^{n-r} = \binom{n}{i} \lambda^{i} (1-\lambda)^{n-i}\,\theta^{\,r-i} \leq \binom{n}{i} \lambda^{i} (1-\lambda)^{n-i}.
  \]
  Summing over $0 \leq i \leq r$ and extending the sum to $0 \leq i \leq n$,
  \[
    \Vol_q(r,n)\,(q-1)^{-r}\lambda^{r}(1-\lambda)^{n-r} \leq \sum_{i=0}^{n}\binom{n}{i} \lambda^{i}(1-\lambda)^{n-i} = (\lambda + (1-\lambda))^{n} = 1,
  \]
  the last step by the binomial theorem. Rearranging gives $\Vol_q(r,n) \leq q^{n H_q(\lambda)}$.
\end{proof}
