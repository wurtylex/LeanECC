\section{Lemmas}

\begin{lemma}[subspace-imp-generator]
    \uses{def:subspace-def}
    \uses{def:generator-def}
    Let $C$ be a linear code that is a subspace of $\mathbb{F}^{n}_{q}$ per subspace-def, then $\exists G$ and $nxn$ matrix such that $C=\{fG | f\in \mathbb{F}^{n}_{q}\}$
\end{lemma}

\begin{proof}
    Suppose we take $G$ to have rows that are just the transverses of the basis vectors of $C$.
    It would follow first of all, that $C$ must have basis vectors since it is a vector space.
    It would further follow that as a vector subspace, it should have no more than $n$ basis vectors (the rest of the rows can be zero).
    From there we could argue that multiplication by all possible $f\in \mathbb{F}^{n}_{q}$ represents all possible linear combinations of the basis vectors and thus the vector space defined by them, i.e. $C$
\end{proof}

\begin{lemma}[generator-imp-subspace]
    \uses{def:subspace-def}
    \uses{def:generator-def}
    Let $G$ an $nxn$ over $\mathbb{F}_{q}$ matrix such that $C=\{fG | f\in \mathbb{F}^{n}_{q}\}$, then $C$ is a subspace of $\mathbb{F}^{n}_{q}$
\end{lemma}

\begin{proof}
    We can show that the zero vector $\vec{0}\in \mathbb{F}^{n}_{q}$ is in $C$ because $\vec{0}G=\vec{0}$
    We can show that $C$ is closed under addition by considering that for any arbitrary $a,b\in C$ it means that
        $\exists x,y\in \mathbb{F}^{n}_{q}$ such that $a=xG,b=yG$
        and from distributivity of matrix multiplication that $a+b=xG+yG=(x+y)G$ which should be in $C$ by definition
    We can show that $C$ is closed under scalar multiplication by considering that for any arbitrary $a\in C$ that
        $\exists x\in \mathbb{F}^{n}_{q}$ such that $a=xG$ ad that therefore for any $d\in \mathbb{F}_{q}$
        $da=dxG$ and since $dx\in \mathbb{F}^{n}_{q}$ it must follow $da\in C$
    These three conditions should be sufficient to prove that $C$ is a valid vector space.
    It is also *tRivIallY* true that because $C$ remains expressed in terms of $\mathbb{F}_{q}$ that $C\subseteq \mathbb{F}^{n}_{q}$
    Therefore $C$ is a subspace of $\mathbb{F}^{n}_{q}$
\end{proof}
